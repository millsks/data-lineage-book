% header.tex — LaTeX preamble for KDP PDF build
\usepackage{graphicx}
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}
\fancyhead[LE]{\leftmark}
\fancyhead[RO]{\rightmark}
\fancyfoot[C]{\thepage}
\renewcommand{\headrulewidth}{0.4pt}
\renewcommand{\footrulewidth}{0pt}

% Ensure chapter-opening pages (plain style) still show page numbers
\fancypagestyle{plain}{%
  \fancyhf{}%
  \fancyfoot[C]{\thepage}%
  \renewcommand{\headrulewidth}{0pt}%
  \renewcommand{\footrulewidth}{0pt}%
}

\usepackage{enumitem}
\setlistdepth{9}

% Better code block formatting — smaller font, line breaking
\usepackage{fvextra}
\DefineVerbatimEnvironment{Highlighting}{Verbatim}{
  fontsize=\footnotesize,
  breaklines=true,
  breakanywhere=true,
  commandchars=\\\{\}
}

% Force a page break after the table of contents
\usepackage{etoolbox}
\apptocmd{\tableofcontents}{\clearpage}{}{}

\raggedbottom

% --- Cover page ---
% Suppress Pandoc's auto-generated \maketitle (metadata is kept for PDF properties)
\renewcommand{\maketitle}{}

% Insert full-bleed cover image as the very first page
\AtBeginDocument{%
  \newgeometry{margin=0pt}
  \begin{titlepage}
    \thispagestyle{empty}
    \noindent
    \includegraphics[width=\paperwidth, height=\paperheight]{resources/book-cover-image.png}
  \end{titlepage}
  \restoregeometry
}
